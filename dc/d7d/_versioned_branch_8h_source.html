72  template < typename TYPE >
73  const TYPE* getValueOrNil ( size_t version, string key ) const {
74 
75  // start searching at the current branch.
76  const VersionedBranch* branch = this;
77 
78  // iterate through parent branches.
79  for ( ; branch; branch = branch->mSourceBranch.get ()) {
80 
81  // ignore branches above the version we're searching for.
82  if ( branch->mVersion <= version ) {
83 
84  // check for a value stack without recursion.
85  const AbstractValueStack* abstractValueStack = branch->findValueStack ( key );
86 
87  if ( abstractValueStack ) {
88 
89  // we can optimize this later if we need to.
90  const ValueStack < TYPE >* valueStack = dynamic_cast < const ValueStack < TYPE >* >( abstractValueStack );
91  if ( !valueStack ) {
92  return NULL; // value exists, but is wrong type.
93  }
94 
95  // check if value stack contains an ID for a matching or most recent lesser version.
96  // it's possible that all values are more recent than the version, in which case
97  // NULL will be returned and we'll keep searching.
98  const TYPE* value = valueStack->getValueOrNil ( version );
99  if ( value ) {
100  return value;
101  }
102  }
103  }
104 
105  // cap the version at the base version before moving to the parent branch.
106  // necessary because branches don't always emerge from the top.
107  version = branch->mVersion;
108  }
109  return NULL;
110  }

124  template < typename TYPE >
125  void setValue ( size_t version, string key, const TYPE& value ) {
126 
127  assert ( this->mVersion <= version );
128 
129  unique_ptr < AbstractValueStack >& abstractValueStack = this->mValueStacksByKey [ key ];
130 
131  if ( !abstractValueStack ) {
132  abstractValueStack = make_unique < ValueStack < TYPE >>();
133  }
134  assert ( abstractValueStack );
135 
136  // we can optimize this later if we need to.
137  ValueStack < TYPE >* valueStack = dynamic_cast < ValueStack < TYPE >* >( abstractValueStack.get ());
138  if ( !valueStack ) throw TypeMismatchOnAssignException ();
139 
140  valueStack->setValue ( version, value );
141 
142  Layer& layer = this->mLayers [ version ];
143  if ( layer.find ( key ) == layer.end ()) {
144  layer.insert ( key );
145  }
146  }
